WAR_SUPPORT_MIN_VB_CHANCE = 0.001
WAR_SUPPORT_MAX_VB_CHANCE = 0.003

war_support_vg_chance = {
	value = local_var:war_support_temp
	divide = 100 #1
	save_temporary_value_as = w #1

	value = WAR_SUPPORT_MIN_VG_CHANCE #0.005
	add = { 
		value     = scope:w #1
		multiply  = scope:w #1
		multiply  = WAR_SUPPORT_MAX_VG_CHANCE #0.015
	}

	multiply = 100 
}

war_support_g_chance = {
    value = local_var:war_support_temp
    divide = 100
    min = 0
    max = 1
    save_temporary_value_as = w

    value = scope:w
    multiply = { value = 1 subtract = scope:w }

    multiply = WAR_SUPPORT_MAX_G_CHANCE
    multiply = 100
}

	save_temporary_value_as = 1w
	
	value = 0
	add = {
		value = scope:1w
		multiply = scope:1w
	}

    multiply = 100 
}

war_support_vb_chance = {
	value = 100
	subtract = local_var:war_support_temp
	divide = 100
	min = 0
	max = 1

		value = scope:inv_w

}
